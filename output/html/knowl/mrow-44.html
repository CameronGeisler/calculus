\begin{align*} \amp = \lim_{h \to 0} \frac{\sqrt{x + h} - \sqrt{x}}{h} \cdot \frac{\sqrt{x + h} + \sqrt{x}}{\sqrt{x + h} + \sqrt{x}}\\ \amp = \lim_{h \to 0} \frac{(x + h) - x}{h(\sqrt{x + h} + \sqrt{x})}\\ \amp = \lim_{h \to 0} \frac{h}{h(\sqrt{x + h} + \sqrt{x})}\\ \amp = \lim_{h \to 0} \frac{1}{\sqrt{x + h} + \sqrt{x}} \amp\amp \text{cancelling the common factor of }\\ \amp = \frac{1}{\sqrt{x} + \sqrt{x}}\\ \amp = \frac{1}{2\sqrt{x}} \end{align*}
