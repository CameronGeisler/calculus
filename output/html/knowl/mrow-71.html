\begin{align*} \frac{dy}{dx} \amp = \lim_{\Delta x \to 0} \frac{\Delta y}{\Delta x}\\ \amp = \lim_{\Delta x \to 0} \frac{\Delta y}{\Delta u} \cdot \frac{\Delta u}{\Delta x} \amp\amp \text{multiplying and dividing by $\Delta u$, and rearranging}\\ \amp = \lim_{\Delta x \to 0} \frac{\Delta y}{\Delta u} \cdot \lim_{\Delta x \to 0} \frac{\Delta u}{\Delta x} \amp\amp \text{by the limit product rule} \end{align*}
